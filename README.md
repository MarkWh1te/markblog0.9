# 本博客基于Django + Bootstrap,支持markdown preview，搜索等功能。
# 使用方法
git clone git@github.com:MarkWh1te/markblog0.9.git

cd markwh1te

virtualenv env 

source env/bin/activate

pip install -r requirements

python manage.py runserver
